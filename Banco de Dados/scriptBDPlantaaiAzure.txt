create table Cliente 
(
	idCliente int primary key,
    nome varchar (80),
    CPF varchar (12),
    email varchar (80),
    senha varchar (40),
    conta char(1),
	check (conta = 'p' or conta ='f'),
    estado char (2),
    cidade varchar (50),
    bairro varchar (50),
    rua varchar (50),
    número int,
    fkProduto int
);

create table Produto
(
	idProduto int primary key,
	tipoDeVaso int,
    check(tipoDeVaso = 1 or tipoDeVaso = 2 or tipoDeVaso = 3),
	regUmidade int,
	regTemperatura int,
	regTemporal datetime,
    fkPlanta int
);

create table Planta
(
	idPlanta int primary key,
    nome varchar (50),
    tempMínima float,
    tempPriQuartil float,
    tempIdeal float,
    tempTerQuartilT float,
    tempMáxima float,
    umiMínima float,
    umiPriQuartil float,
    umiIdeal float,
    umiTerQuartil float,
    umiMáxima float,
    informação varchar (200)
);
